Ejemplo de Animaciones CSS con @Keyframes


.@keyframes mover {
    0% { transform: translateX(0); }
    50% { transform: translateX(100px); }
    100% { transform: translateX(0); }
  }
  
.ejemplo-animacion-1 {
    animation: mover 4s ease-in-out infinite;
  }
